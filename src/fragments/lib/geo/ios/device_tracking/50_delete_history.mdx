func deleteLocationHistory() async {
    do {
        let device = try await Geo.Device.tiedToUserAndDevice()
        try await Amplify.Geo.deleteLocationHistory(for: device)
    } catch let error as Geo.Error {
        print("Failed to delete location history: \(error)")
    } catch {
        print("Unexpected error: \(error)")
    }
}
